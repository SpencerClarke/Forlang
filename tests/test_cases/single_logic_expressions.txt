program single_logic_expressions

	integer::a= 5, b=0
	real::c=4.5, d=0.0
	character(len=4)::e="h", f=""

	if (a) then
		print *, "a"
	end if

	if (b) then
		print *, "b"
	end if

	if (c) then
		print *, "c"
	end if

	if (d) then
		print *, "d"
	end if

	if (e) then
		print *, "e"
	end if

	if (f) then
		print *, "f"
	end if

end program single_logic_expressions
